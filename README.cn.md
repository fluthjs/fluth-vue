const tips$ = stream$.pipe(
  debounce(300),
  map((value) => `debounce: ${value}`),
  filter((value) => value.includes("welcome")),
  map((value) => `filter: ${value}`),
);

  if (words.length > 0) {
    stream$.next(`${stream$.value} ${words.shift()}`);
